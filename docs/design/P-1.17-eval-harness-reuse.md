```typescript
// src/recombination/validation/executor.ts (신규)
import { executeCase as baseExecute } from "../../skills/creator/evals/runner.js";
import type { EvalCase, RawRunResult, EvalConfig } from "../../skills/creator/evals/types.js";
import type { PipelineStrategies } from "../strategy.js";

export interface PlasmidValidationContext {
  readonly plasmidName: string;
  readonly tier: EvalTier;
  readonly workingDirectory: string;
  readonly sessionId: string;
  readonly strategies: PipelineStrategies;
  readonly abortSignal: AbortSignal;
}

export async function executePlasmidCase(
  ec: EvalCase,
  ctx: PlasmidValidationContext,
): Promise<RawRunResult> {
  // base executeCase에 plasmid 전용 context 주입
  const result = await baseExecute(ec, {
    name: "plasmid-validation",
    withSkill: true,                 // plasmid-derived artifact 로드 상태
    model: resolveValidationModel(),  // P-1.21 dual-model config
  } satisfies EvalConfig, {
    workingDirectory: ctx.workingDirectory,
    abortSignal: ctx.abortSignal,
  });

  return {
    ...result,
    runId: `recomb-${ctx.sessionId}/plasmid-${ctx.plasmidName}/tier-${ctx.tier}/case-${ec.id}`,
  };
}
```

  return runParallel(cases, ctx, ctx.strategies.validationParallelism);

async function runSequential(cases, ctx): Promise<RawRunResult[]> {
  const results: RawRunResult[] = [];
  const deadline = Date.now() + (ctx.strategies.validationVolume === "minimal" ? 180_000 : 300_000);

  for (const c of sortByTier(cases)) {
    if (Date.now() > deadline) {
      results.push(markSkipped(c));
      continue;

    results.push(await executePlasmidCase(c, ctx));

      results.push(...cases.slice(results.length).map(markSkipped));
      break;
    }
  }
  return results;
}

## 4. Grader Cascade 상세

```typescript
// src/recombination/validation/grader-cascade.ts (신규)
import type { EvalCase, Grading } from "../../skills/creator/evals/types.js";
import type { RawRunResult } from "../../skills/creator/evals/types.js";
import type { GradingTier } from "../strategy.js";

export async function gradeWithCascade(
  raw: RawRunResult,
  ec: EvalCase,
  tiers: readonly GradingTier[],
): Promise<Grading> {
  const results: GradedExpectation[] = [];

  for (const expectation of ec.expectations) {
    // Priority 1: deterministic
    if (tiers.includes("deterministic")) {
      const det = gradeDeterministic(expectation, raw);
      if (det) { results.push(det); continue; }
    }
    // Priority 2: semi-deterministic
    if (tiers.includes("semi")) {
      const semi = gradeSemi(expectation, raw);
      if (semi) { results.push(semi); continue; }
    }
    // Priority 3: LLM judge (로컬 LLM 시 skip)
    if (tiers.includes("llm")) {
      const llm = await gradeByLlmJudge(expectation, raw);
      results.push(llm);
      continue;
    }
    // Tier에서 해당 grader 제외됨 → skip (not counted)
    results.push({ text: expectation, passed: false, evidence: "skipped (unavailable grader)" });
  }

  return { case_id: ec.id, expectations: results };
}

function gradeDeterministic(expectation: string, raw: RawRunResult): GradedExpectation | null {
  if (expectation.startsWith("output contains ")) {
    const target = extractQuoted(expectation);
    return { text: expectation, passed: raw.output.includes(target), evidence: `contains="${target}"` };
  }
  if (expectation.startsWith("output does NOT contain ")) { /* ... */ }
  if (expectation.startsWith("file:")) { /* FS effect */ }
  if (expectation.startsWith("exit code ")) { /* ... */ }
  return null;   // not handled — pass to next tier
}

function gradeSemi(expectation: string, raw: RawRunResult): GradedExpectation | null {
  if (expectation.startsWith("tool:")) {
    const toolName = expectation.slice(5).split(" ")[0];
    const called = (raw.metrics.tool_calls_by_type[toolName] ?? 0) > 0;
    return { text: expectation, passed: called, evidence: `tool_calls_by_type[${toolName}]=${raw.metrics.tool_calls_by_type[toolName] ?? 0}` };
  }
  if (expectation.startsWith("hook:")) { /* 동일 */ }
  return null;
}
```

// src/recombination/validation/regression-tracker.ts (신규)
import { historySchema, type HistoryEntry } from "../../skills/creator/evals/types.js";
import { appendJsonl, readJsonl } from "../../utils/fs.js";

const VALIDATION_HISTORY_PATH = ".dhelix/recombination/validation-history.jsonl";

export async function appendValidationRun(entry: HistoryEntry): Promise<void> {
  await appendJsonl(VALIDATION_HISTORY_PATH, entry);
}

export async function detectRegressions(current: Benchmark): Promise<Regression[]> {
  const history = await readJsonl(VALIDATION_HISTORY_PATH);
  const lastBenchmark = findPreviousBenchmark(history);
  if (!lastBenchmark) return [];

  return compareBenchmarks(lastBenchmark, current);
}

### Q2 — `createWorkspace` mock level (copy vs full isolation)

**확정**: **Copy-on-write (symlink 우선) + P-1.16 §11 Q4 와 일관**.

구체:
- Artifact 파일 → symlink (Linux/macOS) 또는 copy (Windows)
- Case 실행 중 파일 쓰기 → workspace 내부에 실제 쓰기 (symlink 깨지면 OS가 처리)
- Cleanup: workspace 전체 rm -rf (원본 안 건드림)

**Node.js 구현**:
```typescript
if (process.platform !== "win32" && canSymlink()) {
  await fs.symlink(path.resolve(src), dst);
} else {
  await fs.copyFile(src, dst);
}
```

**Rationale**: Full isolation (in-memory FS) 는 구현 복잡 + 실제 FS 동작 시뮬 부정확. CoW 로 충분.

### Q3 — `ToolContext` plasmid-derived whitelist

**확정**: **P-1.16 §7 `PlasmidValidationContext` 에 `allowedPlasmidTools: readonly string[]` 추가**. Runtime agent 가 실제로 접근 가능한 tool 목록 (artifact 로부터 discover 된) 만 ToolContext 제공.
